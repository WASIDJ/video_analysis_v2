#### 2. 重要性评分算法的精细化
**当前状态**：`significance = range_val * (1 + variance_coef)`
**问题**：过于简单，未考虑动作特异性、关节重要性等因素

**深化方向**：
- **多因素加权评分**：
  ```python
  significance = (
      w1 * normalized_range +          # 变化幅度（归一化）
      w2 * frequency_score +           # 频率重要性

      w4 * anatomical_importance +     # 解剖学重要性
  ```

- **动作类型自适应权重**：不同动作类型使用不同的权重组合
- **专家知识注入**：结合物理治疗师的专业知识

### 附录A：增强特征指纹算法详细设计

#### 1. 频域特征提取
```python
def extract_frequency_features(values, fps):
    """提取频域特征"""
    # FFT变换
    fft_values = np.fft.fft(values)
    freqs = np.fft.fftfreq(len(values), 1/fps)
    
    # 计算特征
    magnitude = np.abs(fft_values)
    power = magnitude ** 2
    
    # 主导频率
    dominant_idx = np.argmax(magnitude[1:len(values)//2]) + 1
    dominant_freq = abs(freqs[dominant_idx])
    
    # 频谱熵
    power_normalized = power / np.sum(power)
    spectral_entropy = -np.sum(power_normalized * np.log2(power_normalized + 1e-10))
    
    # 带宽（频率范围包含90%能量）
    sorted_power = np.sort(power)[::-1]
    cumulative = np.cumsum(sorted_power) / np.sum(power)
    bandwidth_idx = np.where(cumulative >= 0.9)[0][0]
    bandwidth = freqs[bandwidth_idx] - freqs[0]
    
    return {
        "dominant_frequency": dominant_freq,
        "spectral_entropy": spectral_entropy,
        "bandwidth": bandwidth,
        "total_power": np.sum(power),
    }
```

```python
def detect_action_phases(values, fps, min_phase_duration=0.3):
    """检测动作阶段"""
    # 平滑处理
    smoothed = savgol_filter(values, window_length=11, polyorder=3)
    
    # 寻找极值点作为阶段边界
    peaks, _ = find_peaks(smoothed, prominence=0.1)
    valleys, _ = find_peaks(-smoothed, prominence=0.1)
    
    # 合并极值点
    extrema = np.sort(np.concatenate([peaks, valleys]))
    
    # 过滤过短的阶段
    phases = []
    for i in range(len(extrema)-1):
        start = extrema[i]
        end = extrema[i+1]
        duration = (end - start) / fps
        
        if duration >= min_phase_duration:
            phase_values = values[start:end]
            phases.append({
                "start_frame": start,
                "end_frame": end,
                "duration": duration,
                "mean": np.mean(phase_values),
                "range": np.max(phase_values) - np.min(phase_values),
                "type": "peak_to_valley" if i % 2 == 0 else "valley_to_peak"
            })
    
    return phases
```

```python
class FeatureEngineeringPipeline:
    """特征工程管道"""
    
    def __init__(self):
        self.scaler = StandardScaler()
        self.selector = SelectKBest(score_func=f_classif, k=20)
        
    def transform(self, fingerprints):
        """将指纹转换为机器学习特征"""
        features = []
        
        for fp in fingerprints:
            # 提取全局特征
            global_features = [
                fp.total_metrics_analyzed,
                fp.symmetry_score or 0,
                len(fp.active_joints),
            ]
            
            # 提取主导指标特征
            dominant_features = []
            for metric in fp.dominant_metrics[:5]:  # 取前5个主导指标
                dominant_features.extend([
                    metric.mean,
                    metric.std,
                    metric.range,
                    metric.significance_score,
                    metric.variance_coefficient,
                ])
            
            # 填充缺失值
            while len(dominant_features) < 25:  # 5指标 * 5特征
                dominant_features.append(0)
            
            # 合并特征
            sample_features = global_features + dominant_features[:25]
            features.append(sample_features)
        
        # 标准化和特征选择
        features = np.array(features)
        features = self.scaler.fit_transform(features)
        features = self.selector.fit_transform(features, labels)
        
        return features
```

```python
class EnsembleErrorClassifier:
    """集成错误分类器"""
    
    def __init__(self):
        self.models = {
            "rf": RandomForestClassifier(n_estimators=100, random_state=42),
            "xgb": XGBClassifier(n_estimators=100, random_state=42),
            "svm": SVC(probability=True, random_state=42),
        }
        self.meta_model = LogisticRegression()
        
    def fit(self, X, y):
        """训练集成模型"""
        # 第一层：基础模型
        predictions = []
        for name, model in self.models.items():
            model.fit(X, y)
            pred_proba = model.predict_proba(X)
            predictions.append(pred_proba)
        
        # 第二层：元模型
        X_meta = np.hstack(predictions)
        self.meta_model.fit(X_meta, y)
        
    def predict(self, X):
        """预测"""
        predictions = []
        for model in self.models.values():
            pred_proba = model.predict_proba(X)
            predictions.append(pred_proba)
        
        X_meta = np.hstack(predictions)
        return self.meta_model.predict(X_meta)
```
